typedef struct __rbnode_t rbnode_t

A data structure used for a single node in the tree.

It contains a void * pointer to a key, and a void * pointer to an associated value.

Example:

rbnode_t * n = rbtree_find(t, k);
if (n != NULL) {
printf("Found it\n");
printf("Key: %s\n", (char *)rbtree_key(n));
printf("Value: %d\n", *(int *)rbtree_value(n));
} else {
printf("Not found\n");
}

typedef int(* rbtree_compare)(const void *left, const void *right)

A compare function used for insersion and find.

This function should compare the two keys given as const void * pointers, and return:

  • if (left < right) return -1
  • if (left > right) return 1
  • otherwise return 0

See also:

  • rbtree_create()

Example: The following code demonstrates how to create a function that compares two integers passed as (void *) pointers.

int rbnode_data_cmp(const void * left, const void * right)
{
int l = 0;
int r = 0;
l = *(int *)left;
r = *(int *)right;
if (l < r) return(-1);
if (l > r) return(1);
return(0);
}

Example: To use the above compare function, pass the function name as the first argument to rbtree_create().

rbtree_t * t = rbtree_create(rbnode_data_cmp, NULL);
assert(t != NULL);

typedef int(* rbtree_free)(void *key, void *value)

A free function used for node removal.

This function should properly destroy a key and it's associated value, if necessary. It should return 1 on success, and 0 on failure.

See also:

  • rbtree_create()

Example: The following code demonstrates how to create a free function that will be called on all key and value pairs in the tree.

int free_rbnode_data(void * key, void * value) {
free(key);
free(value);
return(1);
}

typedef int(* rbtree_func)(const void *key, void *value)

A processing function used by rbtree_apply() to apply a transformation to each value of each node in the tree.

Expected return value:

  • Success: 1
  • Failure: 0

See also:

  • rbtree_apply()

Example: The following code demonstrates how to create an apply function that will be called on all key/value pairs in a tree.

int apply(void * key, void * value)
{
// 1) Typecast the (void *) value to an (int *)
// 2) De-reference it from an (int *) to an (int)
// 3) Increment it's value by one
(*(int *)value)++;
return(1);
}

typedef struct __rbtree_t rbtree_t

It contains a pointer to the root rbnode_t, a pointer to a compare function, a pointer to a free function, and a size_t value containing the number of nodes in the tree.

Example:

// ... stuff ...
t = rbtree_destroy(t); // t will be set to NULL on success

rbnode_t* rbtree_apply ( rbtree_t *  t,
rbtree_func  func 
)

Apply a transformation to each value of each node in the tree.

Return value:

  • Success: NULL
  • Failure: rbnode_t * pointer to the node on which func() failed

See also:

  • rbtree_func()

Example: The following code demonstrates how to create and use an apply function that will be called on all key/value pairs in a tree.

rbnode_t * n = NULL;
n = rbtree_apply(t, apply);
// On failure, n will point to the node on which the apply function failed
// On success, n will be NULL
assert(n == NULL);
