42 {
43  char * b = (char *)mem_begin(str);

45  size_t capacity = 0;

48  if (str != NULL && b != NULL && e != NULL)

412 dllist_t * str_split(char * src, char * delimiters, size_t max_split)
413 {
414  dllist_t * l = NULL;
415  char * str_b = NULL;
416  char * str_e = NULL;
417  char * str = NULL;
418  size_t c_split = 0;
419 
420  if (src == NULL) return(NULL);
421  if (delimiters == NULL) return(NULL);
422  if (max_split < 1) return(NULL);
423  l = dllist_create(NULL, dllist_free_node);
424  if (l == NULL) return(NULL);
425  str_b = src;
426  str_e = src;
427  while (str_b != NULL && *str_b != '\0' && c_split < max_split-1) {
428  str_e = str_find_char(str_b, delimiters);
429  if (str_e == NULL) {
430  str = str_create(str_b);
431  if (str == NULL) {
432  dllist_destroy(l);
433  return(NULL);
434  }
435  if (dllist_push_back(l, (void *)str) == NULL) {
436  dllist_destroy(l);
437  return(NULL);
438  }
439  }
440  else if (str_e == str_b) {
441  return(l);
442  }
443  else {
444  str = str_substr(str_b, (size_t)(str_e - str_b));
445  if (str == NULL) {
446  dllist_destroy(l);
447  return(NULL);
448  }
449  if (dllist_push_back(l, (void *)str) == NULL) {
450  dllist_destroy(l);
451  return(NULL);
452  }
453  }
454  c_split++;
455  str_e = str_find_not_char(str_e, delimiters);
456  str_b = str_e;
457  }
458  if (str_b != NULL && *str_b != '\0') {
459  str = str_create(str_b);
460  if (str == NULL) {
461  dllist_destroy(l);
462  return(NULL);
463  }
464  if (dllist_push_back(l, (void *)str) == NULL) {
465  dllist_destroy(l);

467  }
468  }
